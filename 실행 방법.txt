## 실행 방법

# 프로젝트 폴더로 이동
cd rpg-fastapi-streamlit

# 가상환경 생성
python -m venv venv

# 활성화 (CMD)
venv\Scripts\activate

# 활성화 (PowerShell)
.\venv\Scripts\Activate.ps1

# 패키지 설치
pip install fastapi uvicorn sqlalchemy pymysql streamlit pandas requests

# 설치 목록 저장
pip freeze > requirements.txt




##AI 모델 학습

#필요 패키지 설치
pip install -r requirements_ai.txt

# Setup
cd rpg_backend
pip install -r ../requirements_ai.txt
copy .env.example .env

#모델 재학습
cd rpg_backend
python -m crafting_ai.train_crafting_model --dataset crafting_ai/data/generated_crafting_dataset_rule_llm_2000.xlsx

#예측 테스트
cd rpg_backend
python -m crafting_ai.test_predictor
