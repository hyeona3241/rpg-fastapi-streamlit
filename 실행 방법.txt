## 실행 방법

# 프로젝트 폴더로 이동
cd rpg-fastapi-streamlit

# 가상환경 생성
python -m venv venv

# 활성화 (CMD)
venv\Scripts\activate

# 활성화 (PowerShell)
.\venv\Scripts\Activate.ps1

# 패키지 설치
pip install fastapi uvicorn sqlalchemy pymysql streamlit pandas requests

# 설치 목록 저장
pip freeze > requirements.txt
